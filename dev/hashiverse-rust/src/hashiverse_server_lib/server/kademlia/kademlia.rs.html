1//! # Server-side Kademlia routing table
2//!
3//! A generic [`Kademlia<ID, PEER>`] DHT that the server uses to answer
4//! `BootstrapV1` / `AnnounceV1` queries and to pick candidate peers for
5//! post-bundle / feedback lookups.
6//!
7//! - **Up to 256 buckets** — one per bit of the id's XOR distance from this
8//!   server's own id.
9//! - **Per-bucket cap** — `max_peers_per_bucket` from
10//!   [`hashiverse_lib::tools::config::SERVER_KADEMLIA_MAX_PEERS_PER_BUCKET`]. When a
11//!   bucket fills, the oldest / least-recently-seen entry is evicted so fresh peers
12//!   always get a slot.
13//! - **Bucket placement** via
14//!   [`hashiverse_lib::tools::tools::LeadingAgreementBits`] — same metric the client
15//!   uses for its `PeerIterator`, so server-side fan-out matches client-side
16//!   expectations.
17//!
18//! Generic over both the id type and the peer struct so tests can instantiate a
19//! simpler `Kademlia` than the production `Kademlia<Id, Peer>`.
20
21use hashiverse_lib::tools::time::TimeMillis;
22use hashiverse_lib::tools::tools;
23use hashiverse_lib::tools::tools::LeadingAgreementBits;
24use std::cmp::min;

26pub trait Peer<ID: AsRef<[u8]> + PartialEq>: Clone {
27    fn id(&self) -> &ID;
28    fn id_hex(&self) -> String {
29        hex::encode(self.id())
30    }
31    fn score(&self, time_millis: TimeMillis) -> f64;
32}
33
34pub struct Kademlia<ID: AsRef<[u8]> + PartialEq, PEER: Peer<ID>> {
35    id: ID,
36    max_peers_per_bucket: usize,
37    peer_buckets: Vec<Vec<PEER>>,
38    peers_last_changed: TimeMillis,
39}
40
41impl<ID: AsRef<[u8]> + PartialEq, PEER: Peer<ID> + std::fmt::Display> Kademlia<ID, PEER> {
42    const MAX_BUCKETS: usize = size_of::<ID>() * 8;
43
44    pub fn new(id: ID, max_peers_per_bucket: usize) -> Self {
45        let mut peer_buckets = Vec::with_capacity(Self::MAX_BUCKETS);
46        for _ in 0..Self::MAX_BUCKETS {
47            peer_buckets.push(Vec::with_capacity(max_peers_per_bucket));
48        }
49
50        Self {
51            id,
52            max_peers_per_bucket,
53            peer_buckets,
54            peers_last_changed: TimeMillis::zero(),
55        }
56    }
57
58    pub fn len(&self) -> usize {
59        let mut len = 0;
60        for peer_bucket in &self.peer_buckets {
61            len += peer_bucket.len();
62        }
63
64        len
65    }
66
67    pub fn is_empty(&self) -> bool {
68        for peer_bucket in &self.peer_buckets {
69            if !peer_bucket.is_empty() {
70                return false;
71            }
72        }
73
74        true
75    }
76
77    pub fn add_peer(&mut self, peer: PEER, time_millis: TimeMillis) -> anyhow::Result<()> {
78        self.peers_last_changed = time_millis;
79        let max_peers_per_bucket = self.max_peers_per_bucket;
80        let peer_id = peer.id();
81        let (peer_bucket, _peer_bucket_id) = self.get_bucket_for_id_mut(peer_id);
82        // trace!("Adding into bucket={} peer={}", peer_bucket_id, peer);
83
84        // Overwrite this peer if we already have it, but only if the new one is 'better'
85        if let Some(i) = peer_bucket.iter().position(|p| *p.id() == *peer_id) {
86            if peer_bucket[i].score(time_millis) < peer.score(time_millis) {
87                // trace!("overwriting peer {}", peer.id_hex());
88                peer_bucket[i] = peer;
89            }
90            return Ok(());
91        }
92
93        // Do we have room for more?
94        if peer_bucket.len() < max_peers_per_bucket {
95            peer_bucket.push(peer);
96            return Ok(());
97        }
98
99        // Is the bucket full?  If so, possibly replace the peer with the lowest score
100        else {
101            let min_score_index = peer_bucket
102                .iter()
103                .enumerate()
104                .min_by(|a, b| a.1.score(time_millis).total_cmp(&b.1.score(time_millis)))
105                .map(|(i, _p)| i);
106
107            if let Some(min_score_index) = min_score_index {
108                if peer_bucket[min_score_index].score(time_millis) < peer.score(time_millis) {
109                    peer_bucket[min_score_index] = peer;
110                }
111            }
112        }
113
114        Ok(())
115    }
116
117    pub fn remove_peer(&mut self, id: &ID, time_millis: TimeMillis) -> Option<PEER> {
118        self.peers_last_changed = time_millis;
119        let (peer_bucket, _) = self.get_bucket_for_id_mut(id);
120        if let Some(i) = peer_bucket.iter().position(|p| *p.id() == *id) {
121            return Some(peer_bucket.swap_remove(i));
122        }
123        None
124    }
125
126    pub fn get_peer_with_lowest_score(&self) -> Option<&PEER> {
127        let mut lowest_score_peer: Option<&PEER> = None;
128        let mut lowest_score: f64 = f64::MAX;
129
130        for peer_bucket in &self.peer_buckets {
131            for peer in peer_bucket {
132                let score = peer.score(self.peers_last_changed);
133                if score < lowest_score {
134                    lowest_score = score;
135                    lowest_score_peer = Some(peer);
136                }
137            }
138        }
139
140        lowest_score_peer
141    }
142
143    pub fn format_bucket_sizes(&self) -> String {
144        format!(
145            "[ {} ]",
146            self.peer_buckets
147                .iter()
148                .map(|bucket| format!("{}", bucket.len()))
149                .collect::<Vec<_>>()
150                .join(", ")
151        )
152    }
153
154    fn get_bucket_for_id(&self, id: &ID) -> &Vec<PEER> {
155        let peer_bucket_id = min(
156            tools::leading_agreement_bits_xor(self.id.as_ref(), id.as_ref()) as usize,
157            Self::MAX_BUCKETS - 1,
158        );
159
160        &self.peer_buckets[peer_bucket_id]
161    }
162
163    fn get_bucket_for_id_mut(&mut self, id: &ID) -> (&mut Vec<PEER>, usize) {
164        let peer_bucket_id = min(
165            tools::leading_agreement_bits_xor(self.id.as_ref(), id.as_ref()) as usize,
166            Self::MAX_BUCKETS - 1,
167        );
168
169        (&mut self.peer_buckets[peer_bucket_id], peer_bucket_id)
170    }
171
172    pub fn get_peers_random(&self, max_peers: usize) -> Vec<PEER> {
173        let mut peers = Vec::with_capacity(max_peers);
174
175        // If we have too few peers, return them all
176        if self.len() <= max_peers {
177            for peer_bucket in &self.peer_buckets {
178                for peer in peer_bucket {
179                    peers.push(peer.clone());
180                }
181            }
182            return peers;
183        }
184
185        // Otherwise return a smattering of peers
186        loop {
187            let peer_bucket = tools::random_element(&self.peer_buckets);
188            if !peer_bucket.is_empty() {
189                let peer = tools::random_element(peer_bucket);
190                peers.push(peer.clone());
191
192                if peers.len() >= max_peers {
193                    return peers;
194                }
195            }
196        }
197    }
198
199    pub fn get_all_peers(&self) -> Vec<PEER> {
200        let mut peers = Vec::with_capacity(self.len());
201        for peer_bucket in &self.peer_buckets {
202            peers.extend(peer_bucket.iter().cloned());
203        }
204        peers
205    }
206
207    pub fn get_peer_buckets(&self) -> &Vec<Vec<PEER>> {
208        &self.peer_buckets
209    }

211    pub fn get_peers_for_key(&self, id: &ID, max_peers: usize) -> (Vec<PEER>, bool) {
212        let peer_bucket = self.get_bucket_for_id(id);
213
214        // Compute (agreement, peer) pairs from the relevant source.
215        // Fast path: all peers in this bucket are provably closer to the target than peers in any other bucket.
216        // Slow path: not enough peers in the target bucket, scan all buckets.
217        let mut agreement_peers: Vec<(LeadingAgreementBits, &PEER)> = if peer_bucket.len() >= max_peers {
218            peer_bucket.iter()
219                .map(|peer| (tools::leading_agreement_bits_xor(id.as_ref(), peer.id().as_ref()), peer))
220                .collect()
221        } else {
222            self.peer_buckets.iter()
223                .flat_map(|bucket| bucket.iter())
224                .map(|peer| (tools::leading_agreement_bits_xor(id.as_ref(), peer.id().as_ref()), peer))
225                .collect()
226        };
227
228        // Sort descending by agreement (closest first)
229        agreement_peers.sort_unstable_by(|a, b| b.0.cmp(&a.0));
230
231        // Find the threshold: include all peers tied with the max_peers-th entry
232        let min_agreement = if agreement_peers.len() <= max_peers {
233            LeadingAgreementBits::MIN
234        } else {
235            agreement_peers[max_peers - 1].0
236        };
237
238        let peers: Vec<PEER> = agreement_peers.iter()
239            .take_while(|(agreement, _)| *agreement >= min_agreement)
240            .map(|(_, peer)| (*peer).clone())
241            .collect();
242        let contains_self = peers.iter().any(|peer| *peer.id() == self.id);
243        (peers, contains_self)
244    }
245
246    pub fn peers_last_changed(&self) -> TimeMillis {
247        self.peers_last_changed
248    }

257    type ID = [u8; 2];
258
259    #[derive(Clone, Debug, PartialEq)]
260    struct Item {
261        id: ID,
262    }
263
264    impl Item {
265        fn new(id: ID) -> Self {
266            Self { id }
267        }
268
269        fn new_random() -> Self {
270            let mut id: ID = [0; size_of::<ID>()];
271            tools::random_fill_bytes(&mut id);
272            Self::new(id)
273        }
274    }
275
276    impl std::fmt::Display for Item {
277        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
278            write!(f, "{}", hex::encode(&self.id))
279        }
280    }
281
282    impl AsRef<[u8]> for Item {
283        fn as_ref(&self) -> &[u8] {
284            &self.id
285        }
286    }
287
288    impl Peer<ID> for Item {
289        fn id(&self) -> &ID {
290            &self.id
291        }
292
293        fn score(&self, _time_millis: TimeMillis) -> f64 { self.id[0] as f64}
294    }
295
296    fn get_closest_peer(id: &Item, peers: &Vec<Item>) -> Result<Item> {
297        let result = peers.iter().max_by(|a, b| {
298            tools::leading_agreement_bits_xor(a.as_ref(), id.as_ref()).cmp(
299                &tools::leading_agreement_bits_xor(b.as_ref(), id.as_ref()),
300            )
301        });
302
303        result
304            .cloned()
305            .ok_or_else(|| anyhow!("should always have a closest peer!"))
306    }

308    #[tokio::test]
309    pub async fn test_get_closest_peers_from_single_bucket() -> Result<()> {
310        for _ in 0..1000 {
311            let item = Item::new_random();
312            trace!("server_id={}", item);
313
314            let mut kademlia = Kademlia::<ID, Item>::new(item.id, 128);
315
316            let num_peers = 64;
317            let mut peers = Vec::with_capacity(num_peers);
318            for _ in 0..num_peers {
319                let peer = Item::new_random();
320                peers.push(peer.clone());
321                kademlia.add_peer(peer.clone(), TimeMillis::zero())?;
322            }
323            trace!("buckets={}", kademlia.format_bucket_sizes());
324
325            for _ in 0..100 {
326                let target_key = Item::new_random();
327                let closest_peer = get_closest_peer(&target_key, &peers)?;
328                let (closest_peers, _) = kademlia.get_peers_for_key(&target_key.id, 1);
329                trace!(
330                    "For target_key={} closest_peer={} closest_peers={}",
331                    target_key,
332                    closest_peer,
333                    tools::format_vec(&closest_peers)
334                );
335
336                let dist1 = tools::leading_agreement_bits_xor(target_key.id(), closest_peer.id());
337                let dist2 = tools::leading_agreement_bits_xor(target_key.id(), closest_peers[0].id());
338                assert_eq!(dist1, dist2);
339            }
340        }
341        Ok(())
342    }
