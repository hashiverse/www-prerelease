9/// Run an async block on the client's dedicated tokio runtime from a blocking
10/// thread, mirroring the Python crate's `py_try!` macro.
11///
12/// Why this exists: napi-rs's `#[napi] async fn` requires the returned future
13/// to be `Send` for all lifetimes — but several `hashiverse_lib::client`
14/// methods produce futures that hold non-`Send`-for-all-lifetimes borrows
15/// across awaits (lock guards over `Arc<dyn …>` trait objects, etc). The
16/// Python crate avoids this by running everything inside `runtime.block_on(...)`
17/// from a thread that has released the GIL; we do the same trick here: spawn a
18/// blocking thread (so the outer future stays trivially `Send`) and `block_on`
19/// the inner work on our dedicated runtime. No code change needed in
20/// `hashiverse_lib`.
21#[macro_export]
22macro_rules! napi_try {
23    ($runtime:expr, { $($body:tt)* }) => {{
24        let runtime = $runtime;
25        ::tokio::task::spawn_blocking(move || {
26            runtime.block_on(async move {
27                anyhow::Ok({ $($body)* })
28            })
29        })
30        .await
31        .map_err($crate::napi_err::join_err_to_napi)?
32        .map_err($crate::napi_err::anyhow_to_napi)
33    }};
34}
