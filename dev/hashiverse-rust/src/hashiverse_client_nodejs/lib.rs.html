15/// Initialise Rust-side logging. Optional. By default Rust log records are silent
16/// in Node; calling `initLogging()` installs `env_logger`, which honours the
17/// `RUST_LOG` environment variable (e.g. `RUST_LOG=info node app.js`).
18///
19/// Process-wide, one-shot. A second call returns the underlying error rather
20/// than silently no-op'ing, matching the Python crate's behaviour.
21#[napi(js_name = "initLogging")]
22pub fn init_logging() -> napi::Result<()> {
23    env_logger::try_init().map_err(|e| napi::Error::from_reason(format!("init_logging failed: {e}")))?;
24    Ok(())
25}
