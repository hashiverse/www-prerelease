1//! Single-threaded PoW search that works on every target.
2//!
3//! No `rayon`, no `spawn_blocking`, no Web Workers — just `pow_generate_with_iteration_limit`
4//! called serially. WASM clients use this directly (with a relaxed `pow_min`); native
5//! callers can use it in unit tests where setting up a thread pool would be overkill.
6//!
7//! Despite the simplicity, this generator does real PoW work — it is not a no-op stub.
8
9use crate::tools::pow::pow_generate_with_iteration_limit;
10use crate::tools::pow_generator::pow_generator::{generate_loop, JobTracker, PowGenerator, PowJobStatus};
11use crate::tools::types::{Hash, Pow, Salt};
12use std::sync::{Arc, Mutex};
13
14pub struct SingleThreadedPowGenerator {
15    tracker: Arc<Mutex<JobTracker>>,
16}
17
18impl SingleThreadedPowGenerator {
19    pub fn new() -> Self {
20        Self { tracker: Arc::new(Mutex::new(JobTracker::default())) }
21    }
22}
23
24impl Default for SingleThreadedPowGenerator {
25    fn default() -> Self { Self::new() }
26}
27
28#[async_trait::async_trait]
29impl PowGenerator for SingleThreadedPowGenerator {
30    async fn generate_best_effort(&self, _label: &str, iteration_limit: usize, pow_min: Pow, data_hash: Hash) -> anyhow::Result<(Salt, Pow, Hash)> {
31        pow_generate_with_iteration_limit(iteration_limit, pow_min, &data_hash).await
32    }
33
34    async fn generate(&self, label: &str, pow_min: Pow, data_hash: Hash) -> anyhow::Result<(Salt, Pow, Hash)> {
35        generate_loop(self, &self.tracker, label, pow_min, data_hash).await
36    }
37
38    fn active_jobs(&self) -> Vec<PowJobStatus> {
39        self.tracker.lock().unwrap().snapshot()
40    }
41
42    fn tracker(&self) -> &Arc<Mutex<JobTracker>> {
43        &self.tracker
44    }
45}

47#[cfg(test)]
48mod tests {
49    use crate::tools::pow::pow_compute_data_hash;
50    use crate::tools::pow_generator::pow_generator::PowGenerator;
51    use crate::tools::pow_generator::single_threaded_pow_generator::SingleThreadedPowGenerator;
52    use crate::tools::tools;
53    use crate::tools::types::Pow;
54
55    #[tokio::test]
56    async fn single_threaded_generates_valid_pow() -> anyhow::Result<()> {
57        const POW_MIN: Pow = Pow(12);
58        let mut data = [0u8; 64];
59        tools::random_fill_bytes(&mut data);
60        let data_hash = pow_compute_data_hash(&[&data]);
61        let generator = SingleThreadedPowGenerator::new();
62        let (_, pow, _) = generator.generate("test", POW_MIN, data_hash).await?;
63        assert!(pow >= POW_MIN);
64        Ok(())
65    }
