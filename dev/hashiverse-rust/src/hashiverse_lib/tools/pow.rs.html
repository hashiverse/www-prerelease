1//! This crate provides a PoW mechanism that should be hugely expensive to reproduce on dedicated hardware.
2//! Given that Hashiverse is predominantly built upon proof of work, we want to put in some effort to make it difficult to cheat as a spammer or sybil using GPU or ASIC advantages.
3//! To this effect we cobble together a chain of different hashing algorithms with different repetition counts - all of which are pseudorandomly chosen based on the initial salt and each subsequent hashing round.
4

14fn apply_hash<H>(data: &Hash) -> anyhow::Result<Hash>
15where H: Digest
16{
17    Hash::from_slice(&H::digest(data.as_ref()).as_slice()[0..32])
18}
19
20fn apply_chained_hash(algo_index: usize, hash_current: Hash) -> anyhow::Result<Hash> {
21
22    const ALGO_COUNT: usize = 17;
23    let algo_index = algo_index % ALGO_COUNT;
24
25    match algo_index {
26        0 => apply_hash::<blake2::Blake2s256>(&hash_current),
27        1 => apply_hash::<blake2::Blake2b512>(&hash_current),
28        2 => apply_hash::<sha2::Sha256>(&hash_current),
29        3 => apply_hash::<sha2::Sha384>(&hash_current),
30        4 => apply_hash::<sha2::Sha512>(&hash_current),
31        5 => apply_hash::<sha3::Sha3_256>(&hash_current),
32        6 => apply_hash::<sha3::Sha3_384>(&hash_current),
33        7 => apply_hash::<sha3::Sha3_512>(&hash_current),
34        8 => apply_hash::<sha3::Keccak256>(&hash_current),
35        9 => apply_hash::<sha3::Keccak384>(&hash_current),
36        10 => apply_hash::<sha3::Keccak512>(&hash_current),
37        11 => apply_hash::<groestl::Groestl256>(&hash_current),
38        12 => apply_hash::<groestl::Groestl512>(&hash_current),
39        13 => apply_hash::<whirlpool::Whirlpool>(&hash_current),
40        14 => apply_hash::<skein::Skein256<U32>>(&hash_current),
41        15 => apply_hash::<skein::Skein512<U64>>(&hash_current),
42
43        // Unfortunately the "digest" trait of the blake3 crate is an older version than we need...so hand roll.
44        16 => {
45            let mut hasher = blake3::Hasher::new();
46            hasher.update(hash_current.as_ref());
47            let hash_output = hasher.finalize();
48            Hash::from_slice(&hash_output.as_bytes()[0..32])
49        },
50
51        _ => Ok(hash_current),
52    }
53}

55/// Pre-hash all input data into a single 32-byte `Hash`.
56///
57/// Call this once before the iteration loop; pass the result to
58/// `pow_measure_from_data_hash` / `pow_generate_with_iteration_limit` so that
59/// workers only receive 32 bytes instead of the full raw data.
60pub fn pow_compute_data_hash(datas: &[&[u8]]) -> Hash {
61    hashing::hash_multiple(datas)
62}
63
64/// Core PoW measurement given an already-pre-hashed data blob.
65///
66/// Computes `hash(data_hash ++ salt)` as the starting point, then runs the
67/// 5-round chained-hash algorithm.  Use `pow_compute_data_hash` to produce
68/// `data_hash` from raw inputs.
69pub fn pow_measure_from_data_hash(data_hash: &Hash, salt: &Salt) -> anyhow::Result<(Pow, Hash)> {
70    let mut data_current = hashing::hash_two(data_hash.as_ref(), salt.as_ref());
71
72    const CHAIN_LENGTH: usize = 5;
73    const MAX_REPETITIONS: usize = 2;
74
75    for _ in 0..CHAIN_LENGTH {
76        let algo_index = data_current.as_bytes()[0] as usize;
77        let repetitions = data_current.as_bytes()[1] as usize % MAX_REPETITIONS;
78
79        for _ in 0..=repetitions {
80            data_current = apply_chained_hash(algo_index, data_current)?;
81        }
82    }
83
84    let leading_zero_bits = tools::count_leading_zero_bits(data_current.as_bytes());
85    Ok((Pow(leading_zero_bits), data_current))
86}
87
88pub fn pow_measure(datas: &[&[u8]], salt: &Salt) -> anyhow::Result<(Pow, Hash)> {
89    pow_measure_from_data_hash(&pow_compute_data_hash(datas), salt)
90}

92/// Try find a sufficient PoW.
93///
94/// It will return the best so far if is it unsuccessful after the `iteration_limit`
95pub async fn pow_generate_with_iteration_limit(iteration_limit: usize, pow_min: Pow, data_hash: &Hash) -> anyhow::Result<(Salt, Pow, Hash)> {
96    let mut salt: Salt;
97    let mut pow_best_so_far = (Salt::zero(), Pow(0), Hash::zero());
98
99    for _ in 0..iteration_limit {
100        salt = Salt::random();
101        let (pow, hash) = pow_measure_from_data_hash(data_hash, &salt)?;
102
103        if pow >= pow_min {
104            return Ok((salt, pow, hash));
105        }
106
107        if pow > pow_best_so_far.1 {
108            pow_best_so_far = (salt, pow, hash);
109        }
110    }
111
112    Ok(pow_best_so_far)
113}

115/// Try forever until a valid PoW is found.
116///
117/// This method "yields" occasionally so that other tokio processes can make progress.
118pub async fn pow_generate(pow_required: Pow, datas: &[&[u8]]) -> anyhow::Result<(Salt, Pow, Hash)> {
119    const BATCH_SIZE: usize = 64 * 1024;
120    let real_time_provider = RealTimeProvider::default();
121    let mut estimator = PowRequiredEstimator::new(real_time_provider.current_time_millis(), "pow_generate", pow_required);
122    let data_hash = pow_compute_data_hash(datas);
123    loop {
124        let result = pow_generate_with_iteration_limit(BATCH_SIZE, pow_required, &data_hash).await?;
125        if result.1 >= pow_required {
126            return Ok(result);
127        }
128
129        let progress = estimator.record_batch_and_estimate(real_time_provider.current_time_millis(), BATCH_SIZE, result.1);
130        trace!("{}", progress);
131        tools::yield_now().await;
132    }
133}

274    /// Regression guard: a future dep bump that silently changes the output of any of the 17
275    /// chained hash algorithms (or of the blake3 pre-hash / hash-two step) will fail this test.
276    ///
277    /// Hard-coded vectors are captured against the current crate versions. When an *intentional*
278    /// future bump changes output, regenerate via the `pow_regression_vectors_print` helper.
279    #[test]
280    fn pow_regression_vectors_match() -> anyhow::Result<()> {
281        for vector in regression_vectors() {
282            let salt_bytes = hex::decode(vector.salt_hex).expect("salt_hex must be valid hex");
283            let salt = Salt::from_slice(&salt_bytes)?;
284            let data_refs: Vec<&[u8]> = vector.datas.iter().map(|v| v.as_slice()).collect();
285
286            let (pow_direct, hash_direct) = pow_measure(&data_refs, &salt)?;
287            let data_hash = pow_compute_data_hash(&data_refs);
288            let (pow_split, hash_split) = pow_measure_from_data_hash(&data_hash, &salt)?;
289
290            assert_eq!(pow_direct, pow_split, "vector {}: two-path mismatch", vector.label);
291            assert_eq!(hash_direct, hash_split, "vector {}: two-path mismatch", vector.label);
292            assert_eq!(pow_direct, Pow(vector.expected_pow), "vector {}: pow drift (likely crypto crate output change)", vector.label);
293            assert_eq!(hex::encode(hash_direct.as_bytes()), vector.expected_final_hash_hex, "vector {}: final-hash drift (likely crypto crate output change)", vector.label);
294        }
295        Ok(())
296    }
